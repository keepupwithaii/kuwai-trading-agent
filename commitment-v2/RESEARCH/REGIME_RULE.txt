# A deterministic, reproducible labelling transform over R1 inputs only. It
# encodes NO market view and NO directional preference: it labels observable
# state, it does not say what to do about it. A sceptic re-derives the label
# bit-for-bit from the same sealed inputs.

inputs: broad-market ETF (IVV) prior-session daily bars only, sealed window.

trend_label =
  up    if close > sma(close, 50) and sma(close,50) > sma(close,200)
  down  if close < sma(close, 50) and sma(close,50) < sma(close,200)
  mixed otherwise

vol_label =
  calm     if realised_vol_20d <  realised_vol_20d_median_1y
  elevated if realised_vol_20d >= realised_vol_20d_median_1y
